-- Bi·∫øn
local enabled = true
local cloning = false
local cloneCount = 0
local plr = game.Players.LocalPlayer

-- Toggle logic
toggle.MouseButton1Click:Connect(function()
    enabled = not enabled
    if enabled then
        toggle.Text = "üü¢ Clone Tool: ON"
        toggle.BackgroundColor3 = Color3.fromRGB(40, 150, 60)
    else
        toggle.Text = "üî¥ Clone Tool: OFF"
        toggle.BackgroundColor3 = Color3.fromRGB(150, 40, 40)
    end
end)

-- Clone tool
local function startCloning(tool)
    if cloning then return end
    cloning = true
    while enabled and tool and tool.Parent == plr.Character do
        local clone = tool:Clone()
        clone.Parent = plr.Backpack
        cloneCount += 1
        countLabel.Text = "üîÅ Clones: " .. tostring(cloneCount)
        task.wait(0.25)
    end
    cloning = false
end

-- Theo d√µi khi nh√¢n v·∫≠t c√≥ tool
local function watchTools(char)
    char.ChildAdded:Connect(function(child)
        if child:IsA("Tool") then
            task.wait(0.1)
            startCloning(child)
        end
    end)
end

-- Khi nh√¢n v·∫≠t xu·∫•t hi·ªán
plr.CharacterAdded:Connect(function(char)
    task.wait(1)
    watchTools(char)
end)

-- N·∫øu nh√¢n v·∫≠t ƒëang c√≥ s·∫µn
if plr.Character then
    watchTools(plr.Character)
    for _, item in ipairs(plr.Character:GetChildren()) do
        if item:IsA("Tool") then
            startCloning(item)
        end
    end
end
